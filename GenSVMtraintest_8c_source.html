304  if (model->epsilon <= 0)

306  break;

309  break;

315  case 'l':
316  model->lambda = atof(argv[i]);
317  if (model->lambda <= 0)
318  exit_invalid_param("lambda", argv);
319  break;

323  strcpy((*model_inputfile), argv[i]);
324  break;
325  case 'm':

328  strcpy((*model_outputfile), argv[i]);
329  break;

334  break;

337  if (model->p < 1.0 || model->p > 2.0)
338  exit_invalid_param("p", argv);

342  break;

358  fprintf(stderr, "Unknown option: -%c\n",

361  }
362  }

367  strcpy((*training_inputfile), argv[i]);
