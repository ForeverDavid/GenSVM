298 double gensvm_percentile(double *values, long N, double p)
299 {
300  if (N == 1)
301  return values[0];
302 
303  long i;
304  double pi, pr, boundary;
305  double *local = Malloc(double, N);
306  for (i=0; i<N; i++)
307  local[i] = values[i];
308 
309  qsort(local, N, sizeof(double), gensvm_dsort);
310  p /= 100.0;
311  p = p*N + 0.5;
312  pi = maximum(minimum(floor(p), N-1), 1);
313  pr = maximum(minimum(p - pi, 1), 0);
314  boundary = (1 - pr)*local[((long) pi)-1] + pr*local[((long) pi)];
315 
316  free(local);
317 
318  return boundary;
319 }
