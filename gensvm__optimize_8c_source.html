151 double gensvm_get_loss(struct GenModel *model, struct GenData *data,
152  struct GenWork *work)
153 {
154  long i, j;
155  long n = model->n;
156  long K = model->K;
157  long m = model->m;
158 
159  double value, rowvalue, loss = 0.0;
160 
161  gensvm_calculate_errors(model, data, work->ZV);
162  gensvm_calculate_huber(model);
163 
164  for (i=0; i<n; i++) {
165  rowvalue = 0;
166  value = 0;
167  for (j=0; j<K; j++) {
168  if (j == (data->y[i]-1))
169  continue;
170  value = matrix_get(model->H, K, i, j);
171  value = pow(value, model->p);
172  rowvalue += value;
173  }
174  rowvalue = pow(rowvalue, 1.0/(model->p));
175  rowvalue *= model->rho[i];
176  loss += rowvalue;
177  }
178  loss /= ((double) n);
179 
180  value = 0;
181  for (i=1; i<m+1; i++) {
182  for (j=0; j<K-1; j++) {
183  value += pow(matrix_get(model->V, K-1, i, j), 2.0);
184  }
185  }
186  loss += model->lambda * value;
187 
188  return loss;

238 void gensvm_calculate_huber(struct GenModel *model)
239 {

242 
243  for (i=0; i<model->n; i++) {
244  for (j=0; j<model->K; j++) {
245  q = matrix_get(model->Q, model->K, i, j);
246  value = 0.0;
247  if (q <= -model->kappa) {
248  value = 1.0 - q - (model->kappa+1.0)/2.0;
