233 long gensvm_kernel_eigendecomp(double *K, long n, double cutoff, double **P_ret,
234  double **Sigma_ret)
235 {
236  int M, status, LWORK, *IWORK = NULL,
237  *IFAIL = NULL;
238  long i, j, num_eigen, cutoff_idx;
239  double max_eigen, abstol, *WORK = NULL,
240  *Sigma = NULL,
241  *P = NULL;
242 
243  double *tempSigma = Malloc(double, n);
244  double *tempP = Malloc(double, n*n);
245 
246  IWORK = Malloc(int, 5*n);
247  IFAIL = Malloc(int, n);
248 
249  // highest precision eigenvalues, may reduce for speed
250  abstol = 2.0*dlamch('S');
251 
252  // first perform a workspace query to determine optimal size of the
253  // WORK array.
254  WORK = Malloc(double, 1);
255  status = dsyevx('V', 'A', 'U', n, K, n, 0, 0, 0, 0, abstol, &M,
256  tempSigma, tempP, n, WORK, -1, IWORK, IFAIL);
257  LWORK = WORK[0];
258 
259  // allocate the requested memory for the eigendecomposition
260  WORK = (double *)realloc(WORK, LWORK*sizeof(double));
261  status = dsyevx('V', 'A', 'U', n, K, n, 0, 0, 0, 0, abstol, &M,
262  tempSigma, tempP, n, WORK, LWORK, IWORK, IFAIL);
263 
264  if (status != 0) {
265  // LCOV_EXCL_START
266  err("[GenSVM Error]: Nonzero exit status from dsyevx.\n");
267  exit(EXIT_FAILURE);
268  // LCOV_EXCL_STOP
269  }
270 
271  // Select the desired number of eigenvalues, depending on their size.
272  // dsyevx sorts eigenvalues in ascending order.
273  max_eigen = tempSigma[n-1];
274  cutoff_idx = 0;
275 
276  for (i=0; i<n; i++) {
277  if (tempSigma[i]/max_eigen > cutoff) {
278  cutoff_idx = i;
279  break;
280  }
281  }
282 
283  num_eigen = n - cutoff_idx;
284 
285  Sigma = Calloc(double, num_eigen);
286  for (i=0; i<num_eigen; i++) {
287  Sigma[i] = tempSigma[n-1 - i];
288  }
289 
290  // revert P to row-major order and copy only the the columns
291  // corresponding to the selected eigenvalues
292  P = Calloc(double, n*num_eigen);
293  for (j=n-1; j>n-1-num_eigen; j--) {
294  for (i=0; i<n; i++) {
295  P[i*num_eigen + (n-1)-j] = tempP[i + j*n];
296  }
297  }
298 
299  free(tempSigma);
300  free(tempP);
301  free(IWORK);
302  free(IFAIL);
303  free(WORK);
304 
305  *Sigma_ret = Sigma;
306  *P_ret = P;
307 
308  return num_eigen;
309 }

424  struct GenData *traindata, double *K2)
425 {

428  *M = NULL;

436

439  for (i=0; i<n1; i++) {

442  matrix_set(M, r, i, j, value);
443  }
444  }
445 
446  // Multiply K2 with M and store in N
447  cblas_dgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, n2, r, n1, 1.0,
448  K2, n1, M, r, 0.0, N, r);
449 
450  // Multiply N with Sigma^{-2}
451  for (j=0; j<r; j++) {
452  value = pow(matrix_get(traindata->Sigma, 1, j, 0), -2.0);
453  for (i=0; i<n2; i++)
454  matrix_mul(N, r, i, j, value);
455  }
456

458  testdata->Z = Calloc(double, n2*(r+1));
459  for (i=0; i<n2; i++) {
460  for (j=0; j<r; j++) {
